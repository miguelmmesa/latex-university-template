%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% FILE: packages.tex
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
% PURPOSE
% --------------
% Loads all LaTeX packages required by the document template.
%
% This file centralizes package management so the template remains
% modular and easy to maintain.
%
%
% ARCHITECTURE ROLE
% ---------------------------------
% Part of the configuration layer of the template.
%
% Responsibilities:
%   - Load packages
%   - Configure package-specific settings
%   - Define bibliography configuration
%
% It MUST NOT contain:
%   - document layout logic
%   - typography rules
%   - visual styles
%
% Those belong to:
%   07_styles/
%
%
% DEPENDENCIES
% ------------------------
% Used by:
%   main.tex
%
% Requires:
%   bibliography file
%   05_bibliography/references.bib
%
%
% NOTES FOR DEVELOPERS
% ---------------------------------------
% Package order can affect compilation.
% In particular:
%
%   hyperref should be loaded near the end
%   cleveref should be loaded after hyperref
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%


% =====================================================
% LANGUAGE AND ENCODING
% =====================================================

\usepackage[spanish]{babel}

\usepackage[T1]{fontenc}

\usepackage[utf8]{inputenc}

% (Optional for modern compilers)
\usepackage[utf8]{inputenc}


% =====================================================
% ENVIROMENTS
% =====================================================

\usepackage{float}

\usepackage[most]{tcolorbox}

\usepackage{verbatim}


% =====================================================
% MATHEMATICS
% =====================================================

\usepackage{amsmath}


% =====================================================
% GRAPHICS AND TABLES
% =====================================================

\usepackage{graphicx}

\usepackage{booktabs}


% =====================================================
% OVERLAY GRAPHICS
% =====================================================

\usepackage{tikz}


% =====================================================
% SOURCE CODE
% =====================================================

\usepackage{listings}
\usepackage{xcolor}

% Rename listings environments (Spanish)
\renewcommand{\lstlistingname}{Código}
\renewcommand{\lstlistlistingname}{Lista de códigos}


% =====================================================
% ALGORITHMS
% =====================================================

\usepackage{algorithm}
\usepackage{algpseudocode}

\floatname{algorithm}{Algoritmo}
\renewcommand{\listalgorithmname}{Lista de algoritmos}


% =====================================================
% FIGURE CAPTIONS
% =====================================================

\usepackage{caption}

\captionsetup{
    font=small,
    labelfont=bf,
    labelsep=period
}


% =====================================================
% ACRONYMS
% =====================================================

\usepackage{acro}

\acsetup{
    first-style = long-short,
    list/name = Lista de Acrónimos
}


% =====================================================
% CITATIONS AND BIBLIOGRAPHY (APA)
% =====================================================

\usepackage{csquotes}

\usepackage[
backend=biber,
style=apa,
sorting=nyt
]{biblatex}

\DeclareLanguageMapping{spanish}{spanish-apa}

\addbibresource{04_bibliography/references.bib}


% =====================================================
% HYPERLINKS
% =====================================================

\usepackage{hyperref}

\hypersetup{
    pdftitle={\documenttitle},
    pdfauthor={\documentauthor},
    pdfcreator={mammesa},
    pdfproducer={mammesa},
    pdfsubject={Trabajo Fin de Grado},
    pdfkeywords={Plantilla,LaTeX},
    colorlinks=true,
    linkcolor=black,
    citecolor=black,
    urlcolor=blue
}


% =====================================================
% INTELLIGENT REFERENCES
% =====================================================

\usepackage{cleveref}


% =====================================================
% PRECISE TIMESTAMP USE
% =====================================================

\usepackage{datetime2}

\typeout{mammesa-template-loaded}
