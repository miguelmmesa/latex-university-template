% --------------------------------------------------
% PAGE GEOMETRY
% --------------------------------------------------

% Controls page margins and printable area

\usepackage[
a4paper,
left=3cm,
right=2.5cm,
top=2.5cm,
bottom=2.5cm,
bindingoffset=0.5cm
]{geometry}



% --------------------------------------------------
% HEADERS AND FOOTERS
% --------------------------------------------------

\usepackage{fancyhdr}

\pagestyle{fancy}

\setlength{\headheight}{16pt}



% --------------------------------------------------
% HEADER / FOOTER CONTENT
% --------------------------------------------------

\fancyhf{}

% Left header → chapter title
\fancyhead[L]{\leftmark}

% Right header → page number
\fancyhead[R]{\thepage}

\renewcommand{\headrulewidth}{0.4pt}



% --------------------------------------------------
% CHAPTER PAGE STYLE
% --------------------------------------------------

% Ensures chapter opening pages follow the same style

\fancypagestyle{plain}{
    \fancyhf{}
    \fancyhead[L]{\leftmark}
    \fancyhead[R]{\thepage}
    \renewcommand{\headrulewidth}{0.4pt}
}
